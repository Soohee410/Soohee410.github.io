 def tagger_time(tagger, texts):
   time_sum = 0
   for sentence in texts:
     t1 = time.time()
     try:
       tagger.morphs(sentence)
     except:
       pass

     t2 = time.time()
     time_sum += (t2 - t1)
   return time_sum
